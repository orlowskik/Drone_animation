41  if (nb_procs>mpi_world_nb_procs()) {
42  if (comm_world_rank==0) {
43  MESSAGE(
44  "Unable to run test: need ", std::to_string(nb_procs), " procs",
45  " but program launched with only ", std::to_string(doctest::mpi_world_nb_procs()), "."
46  );
47  CHECK(nb_procs<=mpi_world_nb_procs());
48  }
49  } else {
50  int color = MPI_UNDEFINED;
51  if(comm_world_rank < nb_procs){
52  color = 0;
53  }
54  MPI_Comm_split(MPI_COMM_WORLD, color, comm_world_rank, &comm);

60  }
