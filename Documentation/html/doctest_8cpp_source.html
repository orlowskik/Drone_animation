323 #ifdef DOCTEST_CONFIG_NO_MULTI_LANE_ATOMICS
324  template <typename T>
325  using AtomicOrMultiLaneAtomic = std::atomic<T>;
326 #else // DOCTEST_CONFIG_NO_MULTI_LANE_ATOMICS
327  // Provides a multilane implementation of an atomic variable that supports add, sub, load,
328  // store. Instead of using a single atomic variable, this splits up into multiple ones,
329  // each sitting on a separate cache line. The goal is to provide a speedup when most
330  // operations are modifying. It achieves this with two properties:
331  //
332  // * Multiple atomics are used, so chance of congestion from the same atomic is reduced.
333  // * Each atomic sits on a separate cache line, so false sharing is reduced.
334  //
335  // The disadvantage is that there is a small overhead due to the use of TLS, and load/store
336  // is slower because all atomics have to be accessed.
337  template <typename T>
338  class MultiLaneAtomic
339  {
340  struct CacheLineAlignedAtomic
341  {
342  std::atomic<T> atomic{};
343  char padding[DOCTEST_MULTI_LANE_ATOMICS_CACHE_LINE_SIZE - sizeof(std::atomic<T>)];
344  };
345  CacheLineAlignedAtomic m_atomics[DOCTEST_MULTI_LANE_ATOMICS_THREAD_LANES];
346 
347  static_assert(sizeof(CacheLineAlignedAtomic) == DOCTEST_MULTI_LANE_ATOMICS_CACHE_LINE_SIZE,
348  "guarantee one atomic takes exactly one cache line");
349 
350  public:
351  T operator++() noexcept { return fetch_add(1) + 1; }
352 
353  T operator++(int) noexcept { return fetch_add(1); }
354 
355  T fetch_add(T arg, std::memory_order order = std::memory_order_seq_cst) noexcept {
356  return myAtomic().fetch_add(arg, order);
357  }
358 
359  T fetch_sub(T arg, std::memory_order order = std::memory_order_seq_cst) noexcept {
360  return myAtomic().fetch_sub(arg, order);
361  }
362 
363  operator T() const noexcept { return load(); }
364 
365  T load(std::memory_order order = std::memory_order_seq_cst) const noexcept {
366  auto result = T();
367  for(auto const& c : m_atomics) {
368  result += c.atomic.load(order);
369  }
370  return result;
371  }
372 
373  T operator=(T desired) noexcept {
374  store(desired);
375  return desired;
376  }
377 
378  void store(T desired, std::memory_order order = std::memory_order_seq_cst) noexcept {
379  // first value becomes desired", all others become 0.
380  for(auto& c : m_atomics) {
381  c.atomic.store(desired, order);
382  desired = {};
383  }
384  }
385 
386  private:
387  // Each thread has a different atomic that it operates on. If more than NumLanes threads
388  // use this, some will use the same atomic. So performance will degrate a bit, but still
389  // everything will work.
390  //
391  // The logic here is a bit tricky. The call should be as fast as possible, so that there
392  // is minimal to no overhead in determining the correct atomic for the current thread.
393  //
394  // 1. A global static counter laneCounter counts continuously up.
395  // 2. Each successive thread will use modulo operation of that counter so it gets an atomic
396  // assigned in a round-robin fashion.
397  // 3. This tlsLaneIdx is stored in the thread local data, so it is directly available with
398  // little overhead.
399  std::atomic<T>& myAtomic() noexcept {
400  static std::atomic<size_t> laneCounter;
401  DOCTEST_THREAD_LOCAL size_t tlsLaneIdx =
402  laneCounter++ % DOCTEST_MULTI_LANE_ATOMICS_THREAD_LANES;
403 
404  return m_atomics[tlsLaneIdx].atomic;
405  }
406  };

557 String& String::operator+=(const String& other) {
558  const unsigned my_old_size = size();
559  const unsigned other_size = other.size();
560  const unsigned total_size = my_old_size + other_size;
561  using namespace std;
562  if(isOnStack()) {
563  if(total_size < len) {
564  // append to the current stack space
565  memcpy(buf + my_old_size, other.c_str(), other_size + 1);
566  // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDeleteLeaks)
567  setLast(last - total_size);
568  } else {
569  // alloc new chunk
570  char* temp = new char[total_size + 1];
571  // copy current data to new location before writing in the union
572  memcpy(temp, buf, my_old_size); // skip the +1 ('\0') for speed
573  // update data in union
574  setOnHeap();
575  data.size = total_size;
576  data.capacity = data.size + 1;
577  data.ptr = temp;
578  // transfer the rest of the data
579  memcpy(data.ptr + my_old_size, other.c_str(), other_size + 1);
580  }
581  } else {
582  if(data.capacity > total_size) {
583  // append to the current heap block
584  data.size = total_size;
585  memcpy(data.ptr + my_old_size, other.c_str(), other_size + 1);
586  } else {
587  // resize
588  data.capacity *= 2;
589  if(data.capacity <= total_size)
590  data.capacity = total_size + 1;
591  // alloc new chunk
592  char* temp = new char[data.capacity];
593  // copy current data to new location before releasing it
594  memcpy(temp, data.ptr, my_old_size); // skip the +1 ('\0') for speed
595  // release old chunk
596  delete[] data.ptr;
597  // update the rest of the union members
598  data.size = total_size;
599  data.ptr = temp;
600  // transfer the rest of the data
601  memcpy(data.ptr + my_old_size, other.c_str(), other_size + 1);
602  }
603  }
604 
605  return *this;
606 }

961  // matching of a string against a wildcard mask (case sensitivity configurable) taken from
962  // https://www.codeproject.com/Articles/1088/Wildcard-string-compare-globbing
963  int wildcmp(const char* str, const char* wild, bool caseSensitive) {
964  const char* cp = str;
965  const char* mp = wild;
966 
967  while((*str) && (*wild != '*')) {
968  if((caseSensitive ? (*wild != *str) : (tolower(*wild) != tolower(*str))) &&
969  (*wild != '?')) {
970  return 0;
971  }
972  wild++;
973  str++;
974  }
975 
976  while(*str) {
977  if(*wild == '*') {
978  if(!*++wild) {
979  return 1;
980  }
981  mp = wild;
982  cp = str + 1;
983  } else if((caseSensitive ? (*wild == *str) : (tolower(*wild) == tolower(*str))) ||
984  (*wild == '?')) {
985  wild++;
986  str++;
987  } else {
988  wild = mp;
989  str = cp++;
990  }
991  }
992 
993  while(*wild == '*') {
994  wild++;
995  }
996  return !*wild;
997  }

2006  void XmlEncode::encodeTo( std::ostream& os ) const {
2007  // Apostrophe escaping not necessary if we always use " to write attributes
2008  // (see: https://www.w3.org/TR/xml/#syntax)
2009 
2010  for( std::size_t idx = 0; idx < m_str.size(); ++ idx ) {
2011  uchar c = m_str[idx];
2012  switch (c) {
2013  case '<': os << "&lt;"; break;
2014  case '&': os << "&amp;"; break;
2015 
2016  case '>':
2017  // See: https://www.w3.org/TR/xml/#syntax
2018  if (idx > 2 && m_str[idx - 1] == ']' && m_str[idx - 2] == ']')
2019  os << "&gt;";
2020  else
2021  os << c;
2022  break;
2023 
2024  case '\"':
2025  if (m_forWhat == ForAttributes)
2026  os << "&quot;";
2027  else
2028  os << c;
2029  break;
2030 
2031  default:
2032  // Check for control characters and invalid utf-8
2033 
2034  // Escape control characters in standard ascii
2035  // see https://stackoverflow.com/questions/404107/why-are-control-characters-illegal-in-xml-1-0
2036  if (c < 0x09 || (c > 0x0D && c < 0x20) || c == 0x7F) {
2037  hexEscapeChar(os, c);
2038  break;
2039  }
2040 
2041  // Plain ASCII: Write it to stream
2042  if (c < 0x7F) {
2043  os << c;
2044  break;
2045  }
2046 
2047  // UTF-8 territory
2048  // Check if the encoding is valid and if it is not, hex escape bytes.
2049  // Important: We do not check the exact decoded values for validity, only the encoding format
2050  // First check that this bytes is a valid lead byte:
2051  // This means that it is not encoded as 1111 1XXX
2052  // Or as 10XX XXXX
2053  if (c < 0xC0 ||
2054  c >= 0xF8) {
2055  hexEscapeChar(os, c);
2056  break;
2057  }
2058 
2059  auto encBytes = trailingBytes(c);
2060  // Are there enough bytes left to avoid accessing out-of-bounds memory?
2061  if (idx + encBytes - 1 >= m_str.size()) {
2062  hexEscapeChar(os, c);
2063  break;
2064  }
2065  // The header is valid, check data
2066  // The next encBytes bytes must together be a valid utf-8
2067  // This means: bitpattern 10XX XXXX and the extracted value is sane (ish)
2068  bool valid = true;
2069  uint32_t value = headerValue(c);
2070  for (std::size_t n = 1; n < encBytes; ++n) {
2071  uchar nc = m_str[idx + n];
2072  valid &= ((nc & 0xC0) == 0x80);
2073  value = (value << 6) | (nc & 0x3F);
2074  }
2075 
2076  if (
2077  // Wrong bit pattern of following bytes
2078  (!valid) ||
2079  // Overlong encodings
2080  (value < 0x80) ||
2081  ( value < 0x800 && encBytes > 2) || // removed "0x80 <= value &&" because redundant
2082  (0x800 < value && value < 0x10000 && encBytes > 3) ||
2083  // Encoded value out of range
2084  (value >= 0x110000)
2085  ) {
2086  hexEscapeChar(os, c);
2087  break;
2088  }
2089 
2090  // If we got here, this is in fact a valid(ish) utf-8 sequence
2091  for (std::size_t n = 0; n < encBytes; ++n) {
2092  os << m_str[idx + n];
2093  }
2094  idx += encBytes - 1;
2095  break;
2096  }
2097  }
2098  }
